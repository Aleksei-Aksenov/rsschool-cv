    def is_palindrome(line: str) -> bool:
    """A function that checks if the given phrase is a palindrome."""

    line = line.replace(' ', '')
    line = line.replace(',', '')
    line = line.replace('.', '')
    line = line.lower()
    check = line[::-1]
    return check == line
